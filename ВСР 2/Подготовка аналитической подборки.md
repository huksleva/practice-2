```julia
using Distributed

# –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –ø—Ä–æ—Ü–µ—Å—Å–æ–≤ (—è–¥–µ—Ä)
addprocs(4)

@everywhere function estimate_pi(n)
    inside = 0
    for i in 1:n
        x, y = rand(), rand()
        inside += (x^2 + y^2 <= 1) ? 1 : 0
    end
    return 4 * inside / n
end

# –†–∞—Å–ø–∞—Ä–∞–ª–ª–µ–ª–∏–≤–∞–Ω–∏–µ –ø–æ —è–¥—Ä–∞–º

    estimate_pi(n_per_worker)
end
```
